  SELECT A.ACCOUNT_NO , COUNT(TRANS_NO) AS COUNT
     FROM T10_TRANS_TABLE A,
          T10_ACCOUNT_TABLE B
     WHERE A.ACCOUNT_NO = B.ACCOUNT_NO
           AND B.ACCOUNT_STATUS = 'A'
     GROUP BY A.ACCOUNT_NO;
